## What Changes and Why

### The Problem With Hardcoded Phenotypes

The current system defines three phenotypes (Caffeine/Sleep, FODMAP, Stress) in:
- `preprocess.py` seed data generation (phenotype labels baked in)
- `src/lib/constants.ts` (CLUSTER_COLORS, CLUSTER_LABELS, AXIS_TO_CLUSTER mappings)
- `src/app/api/chat/route.ts` (SENSITIVITY_AXES definitions, convergence handler cluster mapping)
- `src/components/FlareGraphInner.tsx` (AXIS_TO_CLUSTER for draft node targeting)

In production, HDBSCAN discovers whatever clusters exist in the data. With background features in the vectors, it might find 5-12 clusters that don't map to the three predefined ones. The hardcoded system would either misassign nodes or break.

### The Production Architecture

```
preprocess.py runs:
  1. Generate seed flares (with backgrounds, but NO hardcoded phenotype labels)
  2. Generate + extract narratives (with backgrounds)
  3. Build feature vectors (triggers + background + interactions + embeddings)
  4. HDBSCAN discovers clusters (no preset count)
  5. Claude auto-labels each cluster from its centroid features
  6. UMAP reduces to 3D
  7. Output: flares_processed.json + cluster_metadata.json

At runtime:
  - /api/flares serves both files
  - Frontend reads cluster metadata for colors, labels, descriptions
  - Chat route reads cluster metadata for system prompt
  - No hardcoded phenotype knowledge anywhere in the TypeScript
```

    """Generate a plausible synthetic background based on generation type (A/B/C/mixed)."""
    if generation_type == "mixed" or generation_type not in PHENOTYPE_BACKGROUND_DISTRIBUTIONS:
        generation_type = random.choice(["A", "B", "C"])

    dist = PHENOTYPE_BACKGROUND_DISTRIBUTIONS[generation_type]

        comorbidities.append("anxiety disorder")
    if sex == "female" and random.random() < 0.08:
        comorbidities.append("endometriosis")

    dietary = None
    if random.random() < dist["low_fodmap_chance"]:
        dietary = "low-FODMAP"
    elif random.random() < 0.1:
        dietary = "vegetarian"

    tracks_cycle = sex == "female" and random.random() < 0.4

    return {
        "sex": sex,
        "age_range": age,
        "ibs_subtype": subtype,
        "active_medications": medications,
        "dietary_baseline": dietary,
        "tracks_menstrual_cycle": tracks_cycle,
        "diagnosed_comorbidities": comorbidities,

### 10d: Background Feature Encoding

Add these functions. They produce 8 background features + 4 interaction terms = 12 new dimensions:

```python
# ── Background Feature Encoding ─────────────────────────────────────────────

SEX_ENCODING = {"female": 0.0, "male": 1.0, "other": 0.5, None: 0.5}
AGE_ENCODING = {"18-25": 0.2, "26-35": 0.4, "36-45": 0.6, "46-55": 0.8, "56+": 1.0, None: 0.5}
SUBTYPE_ENCODING = {"IBS-D": 0.0, "IBS-M": 0.5, "IBS-C": 1.0, "unsure": 0.5, None: 0.5}

def build_background_features(background: dict | None) -> np.ndarray:
    """Encode background into 8 normalized features."""
    if background is None:
        return np.full(8, 0.5)

    bg = background
    meds = bg.get("active_medications", [])
    comorbidities = bg.get("diagnosed_comorbidities", [])

    return np.array([
        SEX_ENCODING.get(bg.get("sex"), 0.5),
        AGE_ENCODING.get(bg.get("age_range"), 0.5),
        SUBTYPE_ENCODING.get(bg.get("ibs_subtype"), 0.5),
        float(any("ssri" in m.lower() or "snri" in m.lower() for m in meds)),
        float(any("ppi" in m.lower() for m in meds)),
        float(any("hormonal" in m.lower() or "birth control" in m.lower() for m in meds)),
        float("low-fodmap" in (bg.get("dietary_baseline") or "").lower()),
        float(any("anxiety" in c.lower() for c in comorbidities)),
    ])

BACKGROUND_FEATURE_NAMES = [
    "sex", "age", "ibs_subtype", "on_ssri", "on_ppi",
    "on_hormonal_bc", "low_fodmap_adherent", "anxiety_comorbidity"
]

def build_interaction_features(trigger_features: np.ndarray, bg_features: np.ndarray) -> np.ndarray:
    """Compute 4 trigger×background interaction terms."""
    fodmap_idx = FEATURE_KEYS.index("fodmap_load")
    stress_idx = FEATURE_KEYS.index("stress_level")
    caffeine_idx = FEATURE_KEYS.index("caffeine_before_food")
    # bg indices: ssri=3, age=1, low_fodmap=6, anxiety=7
    return np.array([
        trigger_features[fodmap_idx] * bg_features[6],    # FODMAP × already low-FODMAP
        trigger_features[stress_idx] * bg_features[3],    # stress × on SSRI
        trigger_features[caffeine_idx] * bg_features[1],  # caffeine × age
        trigger_features[stress_idx] * bg_features[7],    # stress × anxiety comorbidity
    ])

INTERACTION_NAMES = [
    "fodmap_x_lowfodmap", "stress_x_ssri", "caffeine_x_age", "stress_x_anxiety"
]
```

### 10e: Updated `build_vectors`

Replace the current function. The vector is now: 16 triggers (×3.0) + 8 background (×1.5) + 4 interactions (×2.0) + 1536 open-text embedding (×1.0) + 1536 narrative embedding (×0.5) = 3100 dimensions.

```python
def build_vectors(flares):
    cached = load_np_cache("vectors.npy")
    if cached is not None:
        return cached

    vecs = []
    for i, flare in enumerate(flares):
        print(f"Embedding {i+1}/{len(flares)}: {flare['id']}")
        ex = flare.get("extracted", {})

        # Layer 1: structured trigger features (16 dims)
        trigger_features = build_layer1(flare)
        layer1 = trigger_features * 3.0

        # Layer 1b: background features (8 dims)
        bg_raw = build_background_features(flare.get("background"))
        layer1b = bg_raw * 1.5

        # Layer 1c: interaction terms (4 dims)
        layer1c = build_interaction_features(trigger_features, bg_raw) * 2.0

        # Layer 2: open-text embedding (1536 dims)
        open_text = " ".join([
            *ex.get("dietary_details", []),
            *ex.get("physiological_details", []),
            *ex.get("psychological_details", []),
            *ex.get("behavioral_details", []),
            *ex.get("novel_factors", [])
        ]).strip()
        layer2 = embed_text(open_text) * 1.0 if open_text else np.zeros(1536)

        # Layer 3: raw narrative embedding (1536 dims)
        raw = flare.get("narrative", "") or ex.get("open_narrative_summary", "")
        layer3 = embed_text(raw) * 0.5 if raw else np.zeros(1536)

        vecs.append(np.concatenate([layer1, layer1b, layer1c, layer2, layer3]))
        time.sleep(0.1)

    matrix = np.array(vecs)
    save_np_cache(matrix, "vectors.npy")
    return matrix
```

### 10f: Dynamic Cluster Labeling via Claude

After HDBSCAN runs, call Claude to generate a clinical label and description for each discovered cluster. This is the key step that replaces hardcoded phenotype definitions.

```python
def label_clusters(flares, labels, feature_matrix):
    """Use Claude to generate clinical labels for each discovered cluster."""
    import anthropic
    anthropic_client = anthropic.Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

    unique_labels = sorted(set(labels))
    cluster_metadata = {}

    # Color palette — enough for up to 15 clusters + noise
    PALETTE = [
        "#FF6B6B", "#4ECDC4", "#FFE66D", "#A78BFA", "#F97316",
        "#06B6D4", "#EC4899", "#84CC16", "#F43F5E", "#8B5CF6",
        "#14B8A6", "#EAB308", "#6366F1", "#D946EF", "#0EA5E9",
    ]

    for cluster_id in unique_labels:
        if cluster_id == -1:
            cluster_metadata[-1] = {
                "cluster_id": -1,
                "label": "UNCLASSIFIED",
                "short_label": "UNCLASSIFIED",
                "description": "Flare patterns that don't match any established cluster. May represent novel trigger combinations.",
                "color": "#888888",
                "count": int(np.sum(labels == -1)),
                "dominant_features": {},
            }
            continue

        # Get indices for this cluster
        member_indices = np.where(labels == cluster_id)[0]
        count = len(member_indices)

        # Compute centroid in structured feature space (first 28 dims: triggers + bg + interactions)
        structured_dims = 16 + 8 + 4  # triggers + background + interactions
        member_vectors = feature_matrix[member_indices, :structured_dims]
        centroid = member_vectors.mean(axis=0)

        # Build readable centroid summary
        all_feature_names = FEATURE_KEYS + BACKGROUND_FEATURE_NAMES + INTERACTION_NAMES
        feature_summary = {}
        for j, name in enumerate(all_feature_names):
            # Undo the weighting to get the raw normalized value
            if j < 16:
                raw_val = centroid[j] / 3.0
            elif j < 24:
                raw_val = centroid[j] / 1.5
            else:
                raw_val = centroid[j] / 2.0
            feature_summary[name] = round(float(raw_val), 2)

        # Collect symptom distribution
        cluster_symptoms = {}
        cluster_backgrounds = []
        for idx in member_indices:
            flare = flares[idx]
            for s in flare.get("extracted", {}).get("symptoms", []):
                cluster_symptoms[s] = cluster_symptoms.get(s, 0) + 1
            if flare.get("background"):
                cluster_backgrounds.append(flare["background"])

        top_symptoms = sorted(cluster_symptoms.items(), key=lambda x: -x[1])[:5]

        # Summarize background distribution
        bg_summary = ""
        if cluster_backgrounds:
            n_bg = len(cluster_backgrounds)
            sex_dist = {}
            for bg in cluster_backgrounds:
                s = bg.get("sex", "unknown")
                sex_dist[s] = sex_dist.get(s, 0) + 1
            sex_str = ", ".join(f"{k}: {round(v/n_bg*100)}%" for k, v in sex_dist.items())

            subtype_dist = {}
            for bg in cluster_backgrounds:
                st = bg.get("ibs_subtype", "unknown")
                subtype_dist[st] = subtype_dist.get(st, 0) + 1
            subtype_str = ", ".join(f"{k}: {round(v/n_bg*100)}%" for k, v in subtype_dist.items())

            anxiety_pct = round(sum(1 for bg in cluster_backgrounds
                if any("anxiety" in c.lower() for c in bg.get("diagnosed_comorbidities", []))) / n_bg * 100)
            ssri_pct = round(sum(1 for bg in cluster_backgrounds
                if any("ssri" in m.lower() or "snri" in m.lower() for m in bg.get("active_medications", []))) / n_bg * 100)
            low_fodmap_pct = round(sum(1 for bg in cluster_backgrounds
                if "low-fodmap" in (bg.get("dietary_baseline") or "").lower()) / n_bg * 100)

            bg_summary = f"""
Background distribution (n={n_bg}):
  Sex: {sex_str}
  IBS subtype: {subtype_str}
  Anxiety comorbidity: {anxiety_pct}%
  On SSRI/SNRI: {ssri_pct}%
  Low-FODMAP adherent: {low_fodmap_pct}%"""

        # Identify dominant features (top 5 that deviate most from 0.5)
        deviations = [(name, abs(feature_summary[name] - 0.5), feature_summary[name])
                      for name in all_feature_names]
        deviations.sort(key=lambda x: -x[1])
        dominant = deviations[:6]
        dominant_str = "\n".join(f"  {name}: {val:.2f}" for name, _, val in dominant)

        # Call Claude to generate label
        labeling_prompt = f"""You are a gastroenterologist analyzing a discovered IBS patient cluster.

Cluster {cluster_id} (n={count} flare events):

Dominant feature values (0-1 scale, 0.5 = average):
{dominant_str}

Top symptoms: {', '.join(f'{s} ({c})' for s, c in top_symptoms)}
{bg_summary}

All feature values:
{json.dumps(feature_summary, indent=2)}

Generate a clinical label for this cluster. Return ONLY this JSON:
{{
  "label": "SHORT UPPERCASE LABEL (max 4 words, e.g. 'FODMAP / STRESS INTERACTION')",
  "short_label": "ABBREVIATED (max 2 words, e.g. 'FODMAP-STRESS')",
  "description": "2-3 sentence clinical description of this endotype. What triggers define it? What background factors are notable? What makes it distinct from other clusters? Reference specific features."
}}"""

        try:
            resp = anthropic_client.messages.create(
                model="claude-sonnet-4-20250514",
                max_tokens=300,
                messages=[{"role": "user", "content": labeling_prompt}]
            )
            text = resp.content[0].text.strip().replace("```json", "").replace("```", "").strip()
            label_data = json.loads(text)
        except Exception as e:
            print(f"Labeling failed for cluster {cluster_id}: {e}")
            label_data = {
                "label": f"CLUSTER {cluster_id}",
                "short_label": f"C{cluster_id}",
                "description": f"Auto-discovered cluster with {count} members.",
            }

        color_idx = cluster_id % len(PALETTE)
        cluster_metadata[cluster_id] = {
            "cluster_id": int(cluster_id),
            "label": label_data["label"],
            "short_label": label_data["short_label"],
            "description": label_data["description"],
            "color": PALETTE[color_idx],
            "count": int(count),
            "dominant_features": {name: val for name, _, val in dominant},
        }

        print(f"  Cluster {cluster_id} ({count} members): {label_data['label']}")
        time.sleep(0.3)

    return cluster_metadata
```

### 10g: Updated `build_output`

The output function now uses dynamic cluster metadata for colors, and includes background summaries:

```python
def summarize_background(bg: dict | None) -> str | None:
    """Short string for hover tooltips."""
    if bg is None:
        return None
    parts = []
    if bg.get("sex"): parts.append(bg["sex"])
    if bg.get("age_range"): parts.append(bg["age_range"])
    if bg.get("ibs_subtype"): parts.append(bg["ibs_subtype"])
    if bg.get("active_medications"): parts.append(", ".join(bg["active_medications"]))
    return " · ".join(parts) if parts else None

def build_output(flares, coords, labels, clusterer, cluster_metadata):
    # Scale UMAP coords (same as current)
    coords_scaled = coords.copy()
    for dim in range(3):
        r = coords_scaled[:, dim].max() - coords_scaled[:, dim].min()
        if r > 0:
            coords_scaled[:, dim] = (coords_scaled[:, dim] - coords_scaled[:, dim].min()) / r
            coords_scaled[:, dim] = (coords_scaled[:, dim] - 0.5) * 200
    coords_scaled[:, 2] *= 0.15

    output = []
    for i, flare in enumerate(flares):
        ex = flare.get("extracted", {})
        symptoms = ex.get("symptoms", [])
        cluster_id = int(labels[i])
        confidence = float(clusterer.probabilities_[i]) if cluster_id != -1 else 0.0

        # Get color from dynamic metadata
        meta = cluster_metadata.get(cluster_id, cluster_metadata.get(-1, {}))
        color = meta.get("color", "#888888")

        output.append({
            "id": flare["id"],
            "x": float(coords_scaled[i][0]),
            "y": float(coords_scaled[i][1]),
            "z": float(coords_scaled[i][2]),
            "clusterId": cluster_id,
            "color": color,
            "confidence": confidence,
            "synthetic": flare.get("synthetic", False),
            "symptoms": symptoms[:3],
            "label": " + ".join(symptoms[:2]) if symptoms else "flare",
            "summary": ex.get("open_narrative_summary", ""),
            "novel_factors": ex.get("novel_factors", []),
            "axis_scores": compute_axis_scores(flare),
            "background_summary": summarize_background(flare.get("background")),
        })

    return output
```

## Step 14: Update `src/hooks/useFlarePolling.ts` — Handle New Response Shape

```typescript
"use client";

import { useEffect, useRef, useState, useCallback } from "react";
import { FlareNode, ClusterMetadata } from "@/lib/types";

export function useFlarePolling(intervalMs = 2000) {
  const [flares, setFlares] = useState<FlareNode[]>([]);
  const [clusterMetadata, setClusterMetadata] = useState<Record<number, ClusterMetadata>>({});
  const [newFlareIds, setNewFlareIds] = useState<Set<string>>(new Set());
  const knownIdsRef = useRef<Set<string>>(new Set());
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const isFirstPollRef = useRef(true);

  const poll = useCallback(async () => {
    try {
      const res = await fetch("/api/flares");
      const data = await res.json();

      // Handle both old format (array) and new format ({ flares, cluster_metadata })
      const flareData: FlareNode[] = Array.isArray(data) ? data : data.flares;
      const metadata = Array.isArray(data) ? {} : (data.cluster_metadata || {});

      const isFirst = isFirstPollRef.current;
      isFirstPollRef.current = false;

      const incoming = new Set<string>();
      for (const node of flareData) {
        if (!knownIdsRef.current.has(node.id) && !isFirst) {
          incoming.add(node.id);
        }
        knownIdsRef.current.add(node.id);
      }

      if (incoming.size > 0 || isFirst) {
        setFlares(flareData);
        setClusterMetadata(metadata);
      }

      if (incoming.size > 0) {
        setNewFlareIds(incoming);
        if (timerRef.current) clearTimeout(timerRef.current);
        timerRef.current = setTimeout(() => {
          setNewFlareIds(new Set());
        }, 3000);
      }
    } catch {
      // silently ignore
    }
  }, []);

  useEffect(() => {
    poll();
    const interval = setInterval(poll, intervalMs);
    return () => {
      clearInterval(interval);
      if (timerRef.current) clearTimeout(timerRef.current);
    };
  }, [poll, intervalMs]);

  return { flares, clusterMetadata, newFlareIds };
}
```
